Firstly, the population in the countries are heavily skewed to the right. Therefore, we can take a look at the median population of a country. import_and_simplify <- function(zip_file) {
  zip_file |>
    unzip() |>
    str_subset(".shp$") |>
    read_sf() |>
    rmapshaper::ms_simplify()
  }
